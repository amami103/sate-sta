/* --- 修正後の checkAnswer 関数 --- */ function checkAnswer() { if (answered) return; const userRaw = document.getElementById('station-input').value.trim(); if (!userRaw) return; // 未入力時は何もしない const userName = userRaw.split(' (')[0]; // 同名駅（同じ名前で場所が違う駅）が複数あるかチェック const sameNameStations = allUniqueStations.filter(s => s.name === userName); // 【追加】同名駅が2つ以上存在し、かつ入力にカッコ（詳細情報）が含まれていない場合 if (sameNameStations.length > 1 && !userRaw.includes(' (')) { alert("同名の駅が複数あります。候補リストから正しい駅（路線や都道府県）を選択してください。"); return; // 回答処理を中断 } let baseScore = 0, timeBonus = 0, dist = 999, isCorrect = false; const time = (Date.now() - startTime) / 1000; // 1. ラベルに完全一致する駅を検索 let userStation = allUniqueStations.find(s => s.fullLabel === userRaw); // 2. 見つからなければ名前だけで検索（同名駅がない場合や、手入力の救済） if (!userStation) { // 同名駅がある場合は上で弾かれているため、ここに来る userName 一致は一意の駅 userStation = sameNameStations.find(s => s.lat === currentStation.lat && s.lng === currentStation.lng) || sameNameStations[0]; } if (userStation) { dist = getDistance(currentStation.lat, currentStation.lng, userStation.lat, userStation.lng); const isSameName = (userName === currentStation.name); // 判定: 「名前一致＋1km以内」 or 「名前不一致でも100m以内（別名駅救済）」 if ((isSameName && dist < 1.0) || dist < 0.1) { isCorrect = true; } } // --- 以下、既存のスコア計算・表示処理 --- let multiplier = 1.0; if (usedHints.pref && usedHints.line) multiplier = 0.1; else if (usedHints.line) multiplier = 0.25; else if (usedHints.pref) multiplier = 0.5; const questionScore = Math.round((baseScore + timeBonus) * multiplier); totalScore += questionScore; const res = document.getElementById('result'); res.innerHTML = `
あなたの解答: ${userRaw}
` + `${isCorrect ? '正解！' : '不正解（正解は「'+currentStation.name+'」）'}` + `
(${currentStation.pref}/${currentStation.line})` + `
解答時間: ${time.toFixed(1)}秒 / 誤差: ${dist > 500 ? '-' : dist.toFixed(2) + 'km'}
` + `
+${questionScore.toLocaleString()}点
`; playHistory.push({ name: currentStation.name, pref: currentStation.pref, line: currentStation.line, userAns: userRaw, isCorrect, dist, score: questionScore, time, lat: currentStation.lat, lng: currentStation.lng }); answered = true; document.getElementById('answer-btn').style.display = "none"; document.getElementById('next-btn').style.display = "block"; questionCount++; }
